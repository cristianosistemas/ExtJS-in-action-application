Ext.ns("TKE.form");


/**
 * @class TKE.form.DepartmentMgrForm
 * @extends Ext.form.FormPanel
 * An Ext.Panel implementation that contains all of the fields to manage departments, including an instance of
 * {@link TKE.grid.EmployeesGridPanel Employees Grid Panel class}.
 * 

 * @constructor
 * @param {Object} config The config object
 * @xtype dept_mgr_form
 **/
TKE.form.DepartmentMgrForm = Ext.extend(Ext.form.FormPanel, {

    //private
    style         : 'border-top: 0px;',

    //private
    layout        : {
        type  : 'vbox',
        align : 'stretch'
    },

    //private
    msgs          : {
        empDisassocCnfrm : 'Are you sure you want to disassociate the selected employees?',
        remDisassocCnfrm : 'Are you sure you want to {0} {1} employees? This action cannot be undone.'
    },

    //private
    initComponent : function() {
        this.employeesGrid = this.buildEmployeesGrid(),
        Ext.apply(this, {
            items : [
                this.buidGeneralInfoForm(),
                this.employeesGrid
            ]
        });

        TKE.form.DepartmentMgrForm.superclass.initComponent.call(this);
        
        this.addEvents({
            
/**
             * @event save
             * Fires when the Save button on top of the panel is clicked.
             * @param {Object} CompanyMgr.DepartmentMgrForm
             * @param {Object} values containing values from the form.
             */
            save            : true,
            
/**
             * @event newemployee
             * Fires when the new employee button on top of the Employees grid panel is clicked.
             * @param {Object} TKE.grid.EmployeesGridPanel
             * @param {Object} CompanyMgr.DepartmentMgrForm
             */
            newemployee     : true,
            
/**
             * @event editemployee
             * Fires when a record in the employees grid panel is clicked.
             * @param {Object} TKE.grid.EmployeesGridPanel
             * @param {Object} CompanyMgr.DepartmentMgrForm
             * @param {Object} Ext.data.Record Containing the data for the row that was double clicked.
             */
            editemployee    : true,
            
/**
             * @event deleteemployee
             * Fires when the delete button in the employees grid panel is clicked.
             * @param {Array} Ext.data.Record Containing the data for the row that was selected when the button was
             *  clicked.
             * @param {Object} TKE.grid.EmployeesGridPanel
             * @param {Object} CompanyMgr.DepartmentMgrForm
             */
            deleteemployee  : true,
            
/**
             * @event unsetdepartment
             * Fires when the delete department button in the top of this panel is clicked.
             * @param {Number} ID The id of the department that is currently being viewed.
             */
            unsetdepartment : true
        });
    },

    //private
    // Constructs the top portion of the panel.
    // @return {Object} A config object to contain the two halves of the form. 
    buidGeneralInfoForm : function() {
        var leftHalf = this.buildGeneralInfoFormLeftHalf();

        var rightHalf = {
            xtype      : 'container',
            title      : 'Description',
            flex       : 1,
            bodyStyle  : 'padding: 1px; margin: 0px;',
            layout     : 'form',
            labelWidth : 70,
            items      : {
                xtype      : 'textarea',
                fieldLabel : 'Description',
                name       : 'description',
                anchor     : '100% 100%'
            }
        } ;

        return {
            tbar         : this.buildTbar(),
            layout       : 'hbox',
            height       : 100,
            bodyStyle    : 'background-color: #DFE8F6; padding: 10px',
            layoutConfig : { align : 'stretch' },
            border       : false,
            items        : [
                leftHalf,               
                rightHalf
            ]
        };
    },

    //private
    // constructs the Top most toolbar
    // @return {Array} An array of {@link Ext.Toolbar.Item}'s
    buildTbar : function() {
        return [
            {
                text    : 'Save',
                iconCls : 'icon-disk',
                scope   : this,
                handler : this.onSave
            },
            {
                text    : 'Reset',
                iconCls : 'icon-arrow_undo',
                scope   : this,
                handler : this.onReset

            },
            '->',
            {
                text    : 'Deactivate Department',
                iconCls : 'icon-delete',
                scope   : this,
                handler : this.onUnsetDepartment
            }
        ];
    },

    //Private
    // Constructs the left half of the department management form.
    // @return {Object} Ext.Container, which contains the form elements.
    buildGeneralInfoFormLeftHalf : function() {
        return {
            xtype       : 'container',
            layout      : 'form',
            flex        : 1,
            labelWidth  : 60,
            defaultType : 'textfield',
            defaults    : { anchor: '-10' },
            items       : [
                {
                    xtype      : 'hidden',
                    name       : 'id'
                },
                {
                    fieldLabel : 'Name',
                    name       : 'name',
                    allowBLank : false,
                    maxLength  : 255
                },
                {
                    xtype      : 'datefield',
                    fieldLabel : 'Activated',
                    name       : 'dateActive'
                }
            ]
        };
    },

    //private
    //Constructs the instance of TKE.grid.EmployeesGridPanel that will sit at the bottom of the screen.
    //@return {Object} TKE.grid.EmployeesGridPanel
    buildEmployeesGrid : function() {
        var textItem = new Ext.Toolbar.TextItem({
            text  : 'Employees',
            style : 'font-weight: bold;'
        });
        var tbar = [
                textItem,
                '->',
            {
                text    : 'New Employee',
                iconCls : 'icon-user_add',
                scope   : this,
                handler : this.onNewEmployee
            },
            '-',
            {
                text    : 'Delete employee',
                iconCls : 'icon-user_delete',
                scope   : this,
                handler : this.onDeleteEmployees
            },
            '-',
            {
                text    : 'Associate Employee(s)',
                iconCls : 'icon-link_add',
                scope   : this,
                handler : this.onAssociateEmployees
            }
        ];

        return new TKE.grid.EmployeesGridPanel({
            flex      : 1,
            loadMask  : true,
            tbar      : tbar,
            style     : 'background-color: #DFE8F6; padding: 10px',
            listeners : {
                scope       : this,
                rowdblclick : this.onGridRowDblClick
            }
        });
    },

    //private
    //Fires the editemployee event.
    onGridRowDblClick : function(grid, rowIndex) {
        var record = grid.store.getAt(rowIndex);

        this.fireEvent('editemployee', this, grid, record);
    },

    
/**
     * A convenience method that retrieve and returns the internal {@link Ext.form.BasicForm}'s values
     * @return {Object} An object containing the department values
     */
    getValues : function() {
        return this.getForm().getValues();
    },

    //private
    //Fires the internal save event.
    onSave : function() {
        if (this.getForm().isValid()) {
            this.fireEvent('save', this, this.getValues());
        }
    },

    //private
    //Resets the form
    onReset : function() {
        if (this.data) {
            this.loadData(this.data);
        }
        else {
            this.getForm().reset();
        }
    },

    
/**
     * Convenience method to create an instance of {@link Ext.data.Record Record} from
     * the internally used {@link TKE.grid.EmployeesGridPanel}
     * @return {Object} Ext.data.Record
     */
    getNewEmployeeRecord : function() {
        return new this.employeesGrid.store.recordType({});
    },

    
/**
     * Convenience method to clear the form's values and internal this.data reference.
     */
    clearForm : function() {
        var vals    = this.getValues();
        var clrVals = {};

        for (var vName in vals)  {
            clrVals[vName] = '';
        }
        
        this.getForm().setValues(clrVals);
        this.data = null;
        this.employeesGrid.removeAll();
    },

    
/**
     * A convenience method to load the internal form and {@link TKE.grid.EmployeesGridPanel} data or clear the entire
     * panel's content.
     * @param {Object} data The data for the form and grid panel.  
     */
    loadData : function(data) {
        if (data) {
            this.data = data;
            this.getForm().setValues(data);
            this.loadEmployeesGrid(data.employees);
        }
        else {
           this.clearForm();
        }
    },

    //private
    //A convenience method to load the internal TKE.grid.EmployeesgridPanel
    loadEmployeesGrid : function(data) {
        this.employeesGrid.loadData(data || []);
    },

    //private
    //Fires the newemployee event.
    onNewEmployee : function() {
        if (this.data) {
            this.fireEvent('newemployee', this, this.employeesGrid);
        }
    },

    //private
    //Opens up a new instance of TKE.window.EmployeeSelectionWindow
    onAssociateEmployees : function() {
        var selectedRecords = this.getEmployeeGridSelections();

        if (this.data && this.data.id) {
            this.empSelectionWindow = new TKE.window.EmployeeSelectionWindow({
                departmentId   : this.data.id,
                departmentName : this.data.name,
                records        : selectedRecords,
                listeners      : {
                    scope   : this,
                    destroy : this.onEmployeeSelectionWindowDestroy
                }
            });
            
            this.relayEvents(this.empSelectionWindow, ['assocemployees']);
            this.empSelectionWindow.show();
        }
    },

    //private a listeners to the internal this.empSelectionWindow destroy event
    onEmployeeSelectionWindowDestroy : function() {
        this.empSelectionWindow = null;
        delete this.empSelectionWindow;
    },
    
    
/**
     * Convenience method to generate a new employee record.
     * @param {Object} o An object for the Ext.data.Record creation.
     */
    genNewEmployeeRecord : function(o) {
        var record = new this.employeesGrid.store.recordType(o);
        this.employeesGrid.store.addSorted(record);
        return record;
    },

    //private
    //Fires the
    onDeleteEmployees : function(btn) {
        var selectedRecs = this.getEmployeeGridSelections();
        if (selectedRecs.length > 0) {
            this.fireEvent('deleteEmployees', selectedRecs, this.employeesGrid, this);
        }
    },

    
/**
     * Convenience method to add records directly to the internal instance of TKE.grid.EmployeesGridPanel 
     * @param {Array} records An array of Ext.data.Records
     */
    addRecordsToEmployeesGrid : function(records) {
        this.employeesGrid.add(records);
    },

    
/**
     * Convenience method to retrieve te list of records selected in the internal instance of TKE.grid.EmployeesGridPanel
     * @return {Array} An array of Ext.data.Records
     */
    getEmployeeGridSelections: function() {
        return this.employeesGrid.selModel.getSelections();
    },

    //private
    //Fires the unsetdepartment event.
    onUnsetDepartment : function() {
        if (this.data) {
            this.fireEvent('unsetdepartment', this.data.id);
        }
    }
});

Ext.reg('department_mgr_form', TKE.form.DepartmentMgrForm);
